181  double sumInvDistance(const bclib::matrix<T> & A)
182  {
183  msize_type nr = A.rowsize();
184  msize_type nc = A.colsize();
185  T oneDistance;
186  T diff;
187  double totalInvDistance = 0.0;
188  /* iterate the row of the first point from 0 to N-2 */
189  for (msize_type irow = 0; irow < nr - 1; irow++)
190  {
191  /* iterate the row the second point from i+1 to N-1 */
192  for (msize_type jrow = (irow + 1); jrow < nr; jrow++)
