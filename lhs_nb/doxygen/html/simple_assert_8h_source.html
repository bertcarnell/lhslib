139  void AssertEqualsLRE(double expected, double actual, int lre, std::string message)
140  {
141  double testlre;
142  if (expected == actual)
143  {
144  return;
145  }
146  if (expected == 0.0)
147  {
148  testlre = -1.0 * log10(abs(actual));
149  }
150  else
151  {
152  testlre = -1.0 * log10(abs(actual-expected)) + log10(abs(expected));
153  }
154 
155  if (isinf(testlre)) // C99 only
156  {
157  return;
158  }
159  if (testlre >= (double) std::numeric_limits<int>::max())
160  {
161  return;
162  }
163  if ((int) floor(testlre) < lre)
164  {
165  std::ostringstream mess;
166  mess << "LRE is " << testlre << ", expected LRE is " << lre << " for " << actual << " vs " << expected;
167  if (!message.empty())
168  {
169  mess << "\n" << message.c_str();
170  }
171  throw assertion_error(mess.str().c_str());
172  }
173  }
